#include <Adafruit_GFX.h>
#include <Adafruit_ST7789.h>
#include <SPI.h>
// ====== Chân TFT (GMT130 - ST7789) ======
//      ST7789   GPIO   
#define TFT_MOSI 6 //SDA
#define TFT_SCLK 4 //SCK
#define TFT_RST  2
#define TFT_DC   0
#define TFT_CS  -1   // Không dùng
Adafruit_ST7789 tft = Adafruit_ST7789(TFT_CS, TFT_DC, TFT_RST);
void setup() {
  Serial.begin(115200);
  delay(500);

  // ====== TFT ======
  SPI.begin(TFT_SCLK, -1, TFT_MOSI, TFT_CS);
  tft.init(240, 240, SPI_MODE3);
  tft.setRotation(0); //xoay màn hình 0,1,2 ,3
  tft.fillScreen(ST77XX_BLUE); //màu nền
  tft.setTextColor(ST77XX_RED); //màu chữ
  tft.setTextSize(3); //size chữ
  tft.setCursor(30, 100);
  tft.println("Hello Thy!");
  tft.setRotation(2);

}
